//Binary to demical in Linkedlist 

int getDecimalValue(ListNode* head) {
     ListNode* temp=head;
     int res=0;
     
     while(temp!=NULL){
         res*=2;
         res+=temp->val;
         temp=temp->next;
     }
        return res;
    }

//Middle of the Linkedlist

 ListNode* middleNode(ListNode* head) {
      if(head==NULL)
      {
          return NULL;
      }
        ListNode* slow=head;
        ListNode* fast=head;
        while(fast!=NULL && fast->next!=NULL)
        {
            fast=fast->next->next;
            slow=slow->next;
        }
        return slow;
    }
